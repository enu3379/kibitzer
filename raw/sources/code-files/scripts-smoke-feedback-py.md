from __future__ import annotations

import sqlite3
import sys
import tempfile
from pathlib import Path

from fastapi.testclient import TestClient

from apps.server.app.config import AppConfig, ControllerConfig, ServerConfig, Tier1Config, Tier2Config
from apps.server.app.main import create_app
from apps.server.app.storage.sqlite import SQLiteStore


def assert_true(condition: bool, message: str) -> None:
    if not condition:
        raise AssertionError(message)


def notify(client: TestClient, title: str, marker: str) -> dict[str, object]:
    request = client.post(
        "/observations/browser-nav",
        json={
            "source": "browser_nav",
            "payload": {
                "url": f"https://example.com/{title.lower().replace(' ', '-')}",
                "title": title,
            },
        },
    ).json()
    assert_true(request["action"] == "request_excerpt", f"{title}: expected request_excerpt")
    result = client.post(
        f"/observations/{request['observation_id']}/excerpt",
        json={"title": title, "text": f"{marker} unrelated page body"},
    ).json()
    assert_true(result["action"] == "notify", f"{title}: expected notify")
    return result


def feedback(client: TestClient, notification: dict[str, object], kind: str) -> dict[str, object]:
    return client.post(
        "/feedback",
        json={
            "kind": kind,
            "intervention_id": notification["intervention_id"],
            "observation_id": notification["observation_id"],
        },
    ).json()


def main() -> int:
    marker = "FEEDBACK_SMOKE_MARKER"
    with tempfile.TemporaryDirectory() as tmp:
        db_path = Path(tmp) / "kibitzer.sqlite3"
        store = SQLiteStore(db_path)
        app = create_app(
            config=AppConfig(
                server=ServerConfig(db_path=str(db_path)),
                tier1=Tier1Config(enabled=False),
                tier2=Tier2Config(enabled=False),
                controller=ControllerConfig(k=1, coldstart_observations=1, cooldown_seconds=0, snooze_seconds=900),
            ),
            store=store,
        )
        with TestClient(app) as client:
            session_id = client.post("/sessions").json()["id"]
            client.post("/sessions/current/goal", json={"raw_text": "Kibitzer observation API"})

            related_notification = notify(client, "Sourdough bread recipe", marker)
            related = feedback(client, related_notification, "related")
            assert_true(related["exemplar_count"] == 2, "related should add one exemplar")

            now_ok = client.post(
                "/observations/browser-nav",
                json={
                    "source": "browser_nav",
                    "payload": {
                        "url": "https://example.com/sourdough-bread-recipe-again",
                        "title": "Sourdough bread recipe",
                    },
                },
            ).json()
            assert_true(now_ok["verdict"] == "OK", "related exemplar should make the same page OK")

            accepted_notification = notify(client, "Weekend hotel rankings", marker)
            accepted = feedback(client, accepted_notification, "accepted")
            assert_true(accepted["intervention_status"] == "accepted", "accepted should mark intervention")

            snooze_notification = notify(client, "Mechanical keyboard deals", marker)
            snoozed = feedback(client, snooze_notification, "snooze")
            assert_true(snoozed["snoozed_until"], "snooze should set snoozed_until")

            blocked = client.post(
                "/observations/browser-nav",
                json={
                    "source": "browser_nav",
                    "payload": {
                        "url": "https://example.com/football-transfer-rumors",
                        "title": "Football transfer rumors",
                    },
                },
            ).json()
            assert_true(blocked["verdict"] == "DRIFT", "post-snooze drift should still be logged")
            assert_true(blocked["action"] == "none", "post-snooze drift should not request excerpt")

        with sqlite3.connect(db_path) as conn:
            feedback_count = conn.execute("SELECT COUNT(*) FROM feedback").fetchone()[0]
            marker_saved = conn.execute(
                "SELECT COUNT(*) FROM event_log WHERE payload_json LIKE ?",
                (f"%{marker}%",),
            ).fetchone()[0]
            final_state = conn.execute(
                "SELECT snoozed_until FROM controller_states WHERE session_id = ?",
                (session_id,),
            ).fetchone()[0]

        assert_true(feedback_count == 3, "three feedback actions should be recorded")
        assert_true(marker_saved == 0, "raw excerpt marker should not be persisted")
        assert_true(final_state is not None, "controller snooze state should persist")

    print("PASS related scenario: feedback added exemplar and next similar page was OK")
    print("PASS accepted scenario: intervention marked accepted")
    print("PASS snooze scenario: next drift was logged but intervention was blocked")
    print("PASS privacy scenario: raw excerpt marker was not persisted")
    return 0


if __name__ == "__main__":
