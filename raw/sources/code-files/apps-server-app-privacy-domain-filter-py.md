from __future__ import annotations

from dataclasses import dataclass
from pathlib import Path
from urllib.parse import urlparse

import yaml


@dataclass(frozen=True)
class SensitiveDomainRules:
    blocked_hosts: list[str]
    blocked_host_keywords: list[str]


@dataclass(frozen=True)
class DropDecision:
    should_drop: bool
    host: str
    reason: str | None = None


def load_sensitive_domain_rules(path: str | Path) -> SensitiveDomainRules:
    config_path = Path(path)
    data = yaml.safe_load(config_path.read_text(encoding="utf-8")) if config_path.exists() else {}
    return SensitiveDomainRules(
        blocked_hosts=list(data.get("blocked_hosts") or []),
        blocked_host_keywords=list(data.get("blocked_host_keywords") or []),
    )


def host_from_url(url: str) -> str:
    return urlparse(url).hostname or ""


def drop_decision_for_url(url: str, rules: SensitiveDomainRules) -> DropDecision:
    parsed = urlparse(url)
    host = parsed.hostname or ""
    normalized_host = host.lower()
    normalized_path = parsed.path or "/"

    for entry in sorted(rules.blocked_hosts, key=len, reverse=True):
        blocked = entry.lower().strip()
        if not blocked:
            continue
        blocked_host, blocked_path = _split_host_path(blocked)
        if _host_matches(normalized_host, blocked_host):
            if blocked_path is None or normalized_path.lower().startswith(blocked_path):
                return DropDecision(True, host, f"blocked_host:{entry}")

    for keyword in rules.blocked_host_keywords:
        normalized_keyword = keyword.lower().strip()
        if normalized_keyword and normalized_keyword in normalized_host:
            return DropDecision(True, host, f"blocked_keyword:{keyword}")

    return DropDecision(False, host)


def should_drop_host(
    host: str,
    blocked_hosts: list[str],
    blocked_keywords: list[str],
) -> bool:
    normalized = host.lower()
    for entry in sorted(blocked_hosts, key=len, reverse=True):
        blocked_host, blocked_path = _split_host_path(entry.lower())
        if blocked_path is None and _host_matches(normalized, blocked_host):
            return True
    return any(keyword.lower() in normalized for keyword in blocked_keywords)


def _split_host_path(entry: str) -> tuple[str, str | None]:
    if "/" not in entry:
        return entry, None
    host, path = entry.split("/", 1)
    return host, f"/{path}".lower()


def _host_matches(host: str, blocked_host: str) -> bool:
    return host == blocked_host or host.endswith(f".{blocked_host}")
